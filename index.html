// Example code for Radiographic Templating Analysis
function calculateTemplate(templateData) {
    const adjustedMeasurements = templateData.map((value) => value * 0.85);
    const sum = adjustedMeasurements.reduce((acc, curr) => acc + curr, 0);
    return sum / adjustedMeasurements.length;
}

// Sample data
const tarsometatarsalData = [24.5, 23.1, 22.8, 24.0];
const averageTemplate = calculateTemplate(tarsometatarsalData);
console.log("Average Templating Measurement: " + averageTemplate);
